FUNCTION prime(n)
	IF n is NOT integer
		return false
	END IF
	IF n < 2
		return false
	END IF
	return ( factorial(n-1)+1 ) mod n == 0
END FUNCTION


FUNCTION print_primes(limit)
	DECLARE primes <- []
	IF limit is NOT integer THEN
		return false
	END IF
	FOR i = 1 to limit
		IF prime(i) == true THEN
			primes.push(i)
		END IF
	END FOR
	OPEN file "./output.txt" in write mode
		FOR each item in primes
			file write item
		END FOR
	CLOSE FILE
END FUNCTION
